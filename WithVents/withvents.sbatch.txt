NPROCS='echo $SLURM_NTASKS'

procs_per_host=1

for host in 'srun hostname | out -d "." -f 1"; do
	if ["$h1" = ""]
	then h1 = "${host}-ib0:${procs_per_host}"
	else h1 = "${h1}-ib0:${procs_per_host}"
	fi
done
